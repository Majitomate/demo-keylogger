// ─────────────────────────────────────────────────
// 1. INTERCEPTACIÓN DE EVENTOS DEL TECLADO
//    Escuchamos el evento "keydown" en el documento
//    completo (o en un elemento específico).
// ─────────────────────────────────────────────────
document.addEventListener('keydown', function(event) {

  // 2. CAPTURA DE LA TECLA
  //    El objeto "event" contiene toda la información:
  const key      = event.key;       // Ej: "a", "Enter", "Shift"
  const keyCode  = event.keyCode;   // Código numérico (legacy)
  const code     = event.code;      // Ej: "KeyA", "Enter"
  const ctrlKey  = event.ctrlKey;   // ¿Ctrl presionado?
  const shiftKey = event.shiftKey;  // ¿Shift presionado?
  const altKey   = event.altKey;    // ¿Alt presionado?

  // 3. TRADUCIR CARACTERES A FORMA LEGIBLE
  //    Las teclas especiales tienen nombres descriptivos.
  const readable = translateKey(key, ctrlKey, shiftKey, altKey);

  // 4. ALMACENAR — acumular en array en memoria
  capturedKeys.push({
    timestamp: new Date().toISOString(),
    key,
    keyCode,
    code,
    readable
  });
});

// ─────────────────────────────────────────────────
// 3. FUNCIÓN DE TRADUCCIÓN A FORMA LEGIBLE
// ─────────────────────────────────────────────────
function translateKey(key, ctrl, shift, alt) {
  const specialKeys = {
    ' ':          '[SPACE]',
    'Enter':     '[ENTER]',
    'Backspace': '[BACKSPACE]',
    'Tab':       '[TAB]',
    'Escape':    '[ESC]',
    'ArrowUp':   '[↑]',
    'ArrowDown': '[↓]',
    'Delete':    '[DEL]',
    'CapsLock':  '[CAPS]',
    // ... más teclas
  };

  let label = specialKeys[key] ?? key;  // usa nombre especial o el char
  if (ctrl)  label = `[CTRL+${label}]`;
  if (alt)   label = `[ALT+${label}]`;
  return label;
}

// ─────────────────────────────────────────────────
// 4. GUARDAR EN ARCHIVO LOCAL — keys.txt
//    Los navegadores no permiten escritura directa
//    al sistema de archivos (seguridad). Se usa la
//    API File System Access o Blob + descarga.
// ─────────────────────────────────────────────────
function downloadLog() {
  const lines = capturedKeys.map(e =>
    `[${e.timestamp}] key="${e.readable}" code=${e.code} keyCode=${e.keyCode}`
  );
  const blob = new Blob([lines.join('\n')], { type: 'text/plain' });
  const url  = URL.createObjectURL(blob);
  const a   = document.createElement('a');
  a.href     = url;
  a.download = 'keys.txt';
  a.click();
  URL.revokeObjectURL(url);
}
